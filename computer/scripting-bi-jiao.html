引數

# python
def func(a, *pargs, **kargs): print a, pargs, kargs
# >>> f(1, 2, 3, x=1, y=2)
# 1 (2, 3) {'y': 2, 'x': 1}
# *pargs: 任意引數 (tuple)
# *kargs: 任意引數 (dict)

# lua
function foo(a, ...)
    print (a)
    print (spam)
end

foo(1,'lovely') -- 1, lovely
